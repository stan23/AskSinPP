95  void process () {
96  int8_t dir = read();
97  if( dir != 0 ) {
98  if( dir < 0 ) dir = -1;
99  else dir = 1;
100  sysclock.cancel(*this);
101  if( last != 0 && last != dir ) {
102  trigger(sysclock);
103  }
104  sdev.channel(dir < 0 ? DownChannel : UpChannel).state(StateButton<>::longpressed);
105  set(millis2ticks(400));
106  last = dir;
