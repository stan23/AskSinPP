238  void decode(uint8_t *buf, uint8_t len) {
239  uint8_t prev = buf[0];
240  buf[0] = (~buf[0]) ^ 0x89;
241  uint8_t i, t;
242  for (i=1; i<len-1; i++) {
243  t = buf[i];
244  buf[i] = (prev + 0xdc) ^ buf[i];
245  prev = t;
246  }
247  buf[i] ^= buf[1];
248  }

255  buf[0] = (~buf[0]) ^ 0x89;
256  uint8_t buf2 = buf[1];

261  buf[i] = prev;
262  }
263  buf[i] ^= buf2;
264  }
